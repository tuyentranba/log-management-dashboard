Create docker-compose.yml that orchestrates postgres, backend, and frontend services with health checks and bind mounts for hot-reload.

**docker-compose.yml** at project root:
```yaml
version: '3.8'

services:
  postgres:
    image: postgres:18-alpine
    container_name: logs-postgres
    environment:
      POSTGRES_DB: logs_db
      POSTGRES_USER: logs_user
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U logs_user -d logs_db"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 10s

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: logs-backend
    command: uvicorn app.main:app --host 0.0.0.0 --port 8000 --reload
    volumes:
      - ./backend:/app:rw
    ports:
      - "8000:8000"
    env_file:
      - .env
    depends_on:
      postgres:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/api/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: logs-frontend
    command: npm run dev
    volumes:
      - ./frontend:/app:rw
      - /app/node_modules
    ports:
      - "3000:3000"
    env_file:
      - .env
    depends_on:
      - backend

volumes:
  postgres_data:
```

**Critical configuration details (per CONTEXT.md and RESEARCH.md):**
- **3 services**: postgres, backend, frontend (exact requirement from CONTEXT.md)
- **Bind mounts**: `./backend:/app:rw` and `./frontend:/app:rw` enable hot-reload per CONTEXT.md
- **Frontend node_modules protection**: `/app/node_modules` anonymous volume prevents local machine's node_modules from overwriting container's (per RESEARCH.md Pitfall 6)
- **Health checks**: postgres uses `pg_isready`, backend uses `curl /api/health` (enables service_healthy condition per RESEARCH.md Pitfall 5)
- **depends_on with condition**: backend waits for `postgres: service_healthy` before starting (prevents connection errors per RESEARCH.md Pitfall 5)
- **env_file**: Both backend and frontend load from `.env` (single source per CONTEXT.md INFRA-04)
- **Named volume**: `postgres_data` persists database between container restarts
- **start_period**: Backend health check waits 40s before first test (allows app startup time)

**Why these specific settings:**
- `service_healthy` condition → backend won't start until postgres is ready, eliminates race condition
- `--reload` flag in uvicorn command → backend auto-restarts on code changes
- `npm run dev` → Next.js fast refresh for frontend hot-reload
- `rw` on bind mounts → allows container to write to host (needed for file generation)

**backend/Dockerfile** (optimized for development with hot-reload):
```dockerfile
FROM python:3.12-slim

WORKDIR /app

# Install system dependencies including curl for health checks
RUN apt-get update && apt-get install -y --no-install-recommends \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements first for layer caching
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy application code (bind mount will override in dev)
COPY . .

# Expose FastAPI port
EXPOSE 8000

# Command is overridden by docker-compose with --reload flag
CMD ["uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000"]
```

**Makefile** at project root:
```makefile
.PHONY: help start stop restart logs logs-backend logs-frontend logs-db test seed migrate clean ps

help:
	@echo "Logs Dashboard - Available Commands"
	@echo "===================================="
	@echo ""
	@echo "  make start       - Start all services (postgres, backend, frontend)"
	@echo "  make stop        - Stop all services"
	@echo "  make restart     - Restart all services"
	@echo "  make logs        - Follow logs from all services"
	@echo "  make logs-backend - Follow backend logs only"
	@echo "  make logs-frontend - Follow frontend logs only"
	@echo "  make logs-db     - Follow database logs only"
	@echo "  make test        - Run backend tests"
	@echo "  make seed        - Populate database with 100k logs"
	@echo "  make migrate     - Run database migrations"
	@echo "  make clean       - Stop services and remove volumes (fresh start)"
	@echo "  make ps          - Show status of all services"
	@echo ""
	@echo "First time setup:"
	@echo "  1. cp .env.example .env"
	@echo "  2. Edit .env with your configuration"
	@echo "  3. make start"
	@echo ""

start:
	@echo "Starting all services..."
	docker-compose up -d
	@echo ""
	@echo "Services started!"
	@echo "  Backend API:  http://localhost:8000"
	@echo "  API Health:   http://localhost:8000/api/health"
	@echo "  API Docs:     http://localhost:8000/docs"
	@echo "  Frontend:     http://localhost:3000"
	@echo ""
	@echo "View logs with: make logs"

stop:
	docker-compose down

restart:
	docker-compose restart

logs:
	docker-compose logs -f

logs-backend:
	docker-compose logs -f backend

logs-frontend:
	docker-compose logs -f frontend

logs-db:
	docker-compose logs -f postgres

test:
	@echo "Running backend tests..."
	docker-compose exec backend pytest tests/ -v

seed:
	@echo "Seeding database with 100k logs (target: <60 seconds)..."
	docker-compose exec backend python scripts/seed.py

migrate:
	@echo "Running database migrations..."
	docker-compose exec backend alembic upgrade head

clean:
	@echo "Stopping services and removing volumes..."
	docker-compose down -v
	@echo "All services stopped and data removed."

ps:
	docker-compose ps
```

**Critical features (per CONTEXT.md requirements):**
- **Single command startup**: `make start` satisfies INFRA-03 requirement
- **Developer ergonomics**: Consistent command interface regardless of platform (per RESEARCH.md Pitfall 8)
- **Commands from CONTEXT.md**: `make start`, `make test`, `make seed` explicitly required
- **Helpful output**: `make start` displays service URLs and next steps
- **Default target**: Running `make` alone shows help menu
- **.PHONY**: Declares targets as commands not files (proper Makefile pattern)

**Why these specific commands:**
- `make start` → Primary command, shows URLs after startup for immediate access
- `make logs-*` → Separate commands for filtering logs by service (debugging productivity)
- `make clean` → Fresh start by removing volumes (useful when database schema changes)
- `make ps` → Quick status check of running services
- `make help` → Self-documenting, reduces "what's the command?" questions
