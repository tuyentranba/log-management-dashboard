```python
@pytest.mark.asyncio
async def test_name(client: AsyncClient, test_db):
    # Create 100k logs
    logs = []
    base_time = datetime(2024, 1, 1, 0, 0, 0, tzinfo=timezone.utc)
    severities = ['INFO', 'WARNING', 'ERROR', 'CRITICAL']

    for i in range(100_000):
        logs.append({
            "timestamp": base_time + timedelta(hours=i // 100),
            "message": f"Log message {i}",
            "severity": severities[i % 4],
            "source": f"service-{i % 5}"
        })

    await test_db.run_sync(
        lambda sync_session: sync_session.bulk_insert_mappings(Log, logs)
    )
    await test_db.commit()

    # Measure endpoint performance
    start = time.perf_counter()
    response = await client.get(f"/api/endpoint?params")
    duration_ms = (time.perf_counter() - start) * 1000

    # Assertions with hard thresholds
    assert response.status_code == 200
    assert duration_ms < threshold_ms, f"Query took {duration_ms:.2f}ms (target: <{threshold_ms}ms)"
```
