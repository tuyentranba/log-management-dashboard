```python
class LogCreate(BaseModel):
    timestamp: datetime  # ISO 8601 with timezone
    message: str  # min_length=1
    severity: str  # Pattern: INFO|WARNING|ERROR|CRITICAL
    source: str  # min_length=1, max_length=100

class LogResponse(BaseModel):
    id: int
    timestamp: datetime
    message: str
    severity: str
    source: str

class LogListResponse(BaseModel):
    data: list[LogResponse]
    next_cursor: Optional[str]  # Base64-encoded cursor token
    has_more: bool
```

```typescript
export interface LogResponse {

  severity: Severity
  source: string
}

export interface LogListResponse {
  data: LogResponse[]
  next_cursor: string | null  // Base64-encoded cursor token
  has_more: boolean
}

export interface LogCreate {
```
